Wrote program that checks to see if I did a Linked List question correctly on the foundation exam. Goal of program is to insert(recursively) a node with data -1 infront of every even node. Example is commented out.

#include stdio.h
#include stdlib.h
// 0 -> 1 -> 2 -> 3
// 0 -> -1 -> 1 -> 2 -> -1 -> 3
typedef struct node
{
int data;
struct node *next;
}node;
void *insert(node *head)
{
if (head == NULL)
return NULL;
node *temp = head, *temp2 = head->next;
head = head->next;
if (temp->data % 2 == 0)
{
temp->next = malloc(sizeof(node));
temp = temp->next;
temp->data = -1;
temp->next = temp2;
}
temp2 = insert(head);
}
node *initialize(node *head, int val)
{
if (head == NULL)
return NULL;
node *temp = NULL;
for (temp = head; temp->next != NULL; temp = temp->next)
;
temp->next = malloc(sizeof(node));
temp = temp->next;
temp->data = val;
return head;
}
node *createList(node *head)
{
head = malloc(sizeof(node));
head->data = 0;
return head;
}
void printList(node *head)
{
node *temp = head;
while (temp != NULL)
{
printf(" %d %s", temp->data, (temp->next == NULL) ? " " : "->");
temp = temp->next;
}
}
int main(void)
{
node *head = createList(head);
int i;
for (i = 1; i <= 4; i++)
head = initialize(head,i);
printf("Before:\n");
printList(head);
insert(head);
printf("\nAfter:\n");
printList(head);
printf("\n");
return 0;
}
